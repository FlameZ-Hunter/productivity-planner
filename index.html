// script.js // Array of motivational quotes const quotes = [ "The best time to start was yesterday. The next best time is now.", "Discipline is the bridge between goals and accomplishment.", "Focus on your goals, not your fears.", "Consistency is what transforms average into excellence.", "You donâ€™t have to be great to start, but you have to start to be great." ]; // Function to show a random quote function showRandomQuote() { const quoteElement = document.getElementById("quote"); const randomIndex = Math.floor(Math.random() * quotes.length); quoteElement.textContent = quotes[randomIndex]; } // Show a quote when the page loads window.onload = function () { showRandomQuote(); // Load saved morning goal if exists const savedMorning = localStorage.getItem("morningGoal"); if (savedMorning) { document.getElementById("morningInput").value = savedMorning; } // Load saved workout log if exists const savedWorkout = localStorage.getItem("workoutLog"); if (savedWorkout) { document.getElementById("workoutInput").value = savedWorkout; } }; // Save morning goal button document.getElementById("saveMorning").addEventListener("click", () => { const morningGoal = document.getElementById("morningInput").value.trim(); localStorage.setItem("morningGoal", morningGoal); alert("Morning goal saved!"); }); // Save workout log button document.getElementById("saveWorkout").addEventListener("click", () => { const workoutLog = document.getElementById("workoutInput").value.trim(); localStorage.setItem("workoutLog", workoutLog); alert("Workout log saved!"); }); // New Quote button document.getElementById("newQuote").addEventListener("click", showRandomQuote);
